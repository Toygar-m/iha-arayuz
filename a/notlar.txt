        MainWindow.setCentralWidget(self.centralwidget)
        self.timer = QtCore.QTimer()
        self.timer.timeout.connect(self.guncelle)
        self.timer.start(30)
        self.cap = cv2.VideoCapture(0)

    def guncelle(self):
        ret, frame = self.cap.read()
        if not ret:
            return

        rgbGoruntu = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        h, w, ch = rgbGoruntu.shape
        bytes_per_line = ch * w
        image = QtGui.QImage(rgbGoruntu.data, w, h, bytes_per_line, QtGui.QImage.Format_RGB888)
        self.label_38.setPixmap(QtGui.QPixmap.fromImage(image))

    def closeEvent(self, event):
        self.cap.release()
